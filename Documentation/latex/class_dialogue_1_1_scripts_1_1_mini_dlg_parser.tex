\doxysubsection{Detailed Description}
Parser for a lightweight dialogue scripting format that produces a set of named \doxylink{class_mini_node}{Mini\+Node} objects. 

The input text is interpreted as a sequence of nodes, each with a {\ttfamily title\+:} header and a body. Nodes are stored in a dictionary keyed by node name (case-\/insensitive). 

{\bfseries{Node structure}}


\begin{DoxyCode}{0}
\DoxyCodeLine{@subsection\ autotoc\_md0\ title:\ Intro}
\DoxyCodeLine{}
\DoxyCodeLine{Hello\ there!}
\DoxyCodeLine{-\/>\ Option\ A\ [\textcolor{keywordflow}{if}\ hasKey]}
\DoxyCodeLine{\ \ \ \ <<jump\ NextNode>>}
\DoxyCodeLine{-\/>\ Option\ B}
\DoxyCodeLine{\ \ \ \ <<\textcolor{keyword}{set}\ hasKey\ =\ \textcolor{keyword}{true}>>}
\DoxyCodeLine{<<\textcolor{keywordflow}{if}\ hasKey>>}
\DoxyCodeLine{\ \ \ \ <<jump\ HasKeyNode>>}
\DoxyCodeLine{<<\textcolor{keywordflow}{else}>>}
\DoxyCodeLine{\ \ \ \ <<jump\ NoKeyNode>>}
\DoxyCodeLine{@section\ autotoc\_md1\ <<endif>>}

\end{DoxyCode}


{\bfseries{Body rules}}


\begin{DoxyItemize}
\item Plain lines become \doxylink{class_line_step}{Line\+Step}. 
\item Lines starting with {\ttfamily -\/\texorpdfstring{$>$}{>}} begin a choice block and become a \doxylink{class_choice_step}{Choice\+Step}. 
\item Command lines are wrapped in {\ttfamily \texorpdfstring{$<$}{<}\texorpdfstring{$<$}{<} ... \texorpdfstring{$>$}{>}\texorpdfstring{$>$}{>}} and become \doxylink{class_command_step}{Command\+Step}. 
\item {\ttfamily \texorpdfstring{$<$}{<}\texorpdfstring{$<$}{<}if ...\texorpdfstring{$>$}{>}\texorpdfstring{$>$}{>}} begins a multi-\/line conditional block ending at {\ttfamily \texorpdfstring{$<$}{<}\texorpdfstring{$<$}{<}endif\texorpdfstring{$>$}{>}\texorpdfstring{$>$}{>}}. 
\end{DoxyItemize}
